<?php
if ($_SERVER['REQUEST_METHOD']==='POST' && !empty($_POST['numbers'])) {
    $input = $_POST['numbers'];
    // 支持空格、逗号、+号分隔
    $nums = preg_split('/[\s,，\+]+/', trim($input), -1, PREG_SPLIT_NO_EMPTY);
    $nums = array_map('intval', $nums);

    // 对应索引显示
    $indexText = "";
    foreach ($nums as $i => $n) {
        $indexText .= ($i+1) . " = " . $n . "\n";
    }

    // 分组显示（相同数字在一行）
    $groups = [];
    foreach ($nums as $i => $n) {
        $pos = $i + 1;
        if (!isset($groups[$n])) $groups[$n] = [];
        $groups[$n][] = $pos;
    }
    ksort($groups);
    $groupText = "";
    foreach ($groups as $num => $positions) {
        $groupText .= $num . " = " . implode(", ", $positions) . "\n";
    }
    echo '<div class="result"><strong>对应索引结果：</strong>'."\n".$indexText.'</div>';
    echo '<div class="result"><strong>分组显示结果：</strong>'."\n".$groupText.'</div>';
}
?>
